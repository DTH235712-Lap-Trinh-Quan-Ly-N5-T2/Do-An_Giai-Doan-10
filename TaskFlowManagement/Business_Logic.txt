            int taskId,
            string sourcePath,
            int requesterId,
            IList<string> requesterRoles)
        {
            var isAuth = await IsAuthorizedForTaskAsync(taskId, requesterId, requesterRoles);
            if (!isAuth)
                return (false, "Báº¡n khÃ´ng cÃ³ quyá»n Ä‘Ã­nh kÃ¨m file trong cÃ´ng viá»‡c nÃ y.", null);

            var fileInfo = new FileInfo(sourcePath);
            if (!fileInfo.Exists)
                return (false, "File khÃ´ng tá»“n táº¡i.", null);

            // Validation theo yÃªu cáº§u Auditor: Kiá»ƒm tra Ä‘uÃ´i file vÃ  dung lÆ°á»£ng
            var extension = fileInfo.Extension.ToLower();
            var blacklisted = new[] { ".exe", ".msi", ".bat", ".cmd", ".ps1" };
            if (blacklisted.Contains(extension))
                return (false, $"Há»‡ thá»‘ng khÃ´ng cho phÃ©p táº£i lÃªn tá»‡p cÃ³ Ä‘á»‹nh dáº¡ng '{extension}' vÃ¬ lÃ½ do báº£o máº­t.", null);

            const long maxFileSize = 10 * 1024 * 1024; // 10MB
            if (fileInfo.Length > maxFileSize)
                return (false, $"Tá»‡p quÃ¡ lá»›n. Dung lÆ°á»£ng tá»‘i Ä‘a cho phÃ©p lÃ  10MB (Tá»‡p hiá»‡n táº¡i: {fileInfo.Length / 1024 / 1024:F1}MB).", null);

            var fileName = fileInfo.Name;
            var sizeBytes = fileInfo.Length;

            // Define upload path
            var uploadsFolder = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Uploads", "Tasks", taskId.ToString());
            if (!Directory.Exists(uploadsFolder))
            {
                Directory.CreateDirectory(uploadsFolder);
            }

            // Generate unique filename to avoid collision
            var uniqueFileName = $"{Guid.NewGuid()}_{fileName}";
            var destinationPath = Path.Combine(uploadsFolder, uniqueFileName);

            try
            {
                await Task.Run(() => File.Copy(sourcePath, destinationPath, true));
            }
            catch (Exception ex)
            {
                return (false, $"Lá»—i khi táº£i lÃªn tá»‡p: {ex.Message}", null);
            }

            var relativePath = Path.Combine("Uploads", "Tasks", taskId.ToString(), uniqueFileName);

            var attachment = new Attachment
            {
                TaskItemId = taskId,
                UploadedById = requesterId,
                FileName = fileName,
                FilePath = relativePath,
                ContentType = extension,
                FileSizeBytes = sizeBytes,
                UploadedAt = DateTime.UtcNow
            };

            await _attachmentRepo.AddAsync(attachment);

            var addedAttachment = await _attachmentRepo.GetByIdAsync(attachment.Id);

            NotifyDataChanged();
            return (true, "Táº£i tá»‡p thÃ nh cÃ´ng.", addedAttachment);
        }
